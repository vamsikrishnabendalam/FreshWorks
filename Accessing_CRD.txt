Accessing CRD
>>> import CRD as i                # imports CRD into i variable
>>> i.create("vamsi",13)           # Creates Dictionary
>>> i.read("vamsi")                # Reads   Dictionary
>>> i.delete("vamsi")              # Deletes Dictionary
>>> i.create("Krishna",2000,60)    # Creates Dictionary with certain time period (1=1second)
